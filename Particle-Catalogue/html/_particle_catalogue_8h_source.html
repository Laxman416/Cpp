65template<class ParticleType>
66void ParticleCatalogue<ParticleType>::add_particle(const SafeSharedPtr<ParticleType>& particle)
67{
68 std::string particle_type;
69 std::string key;
70 if(particle->get_name() == "null")
71 {
72 particle_type = "null";
73 key = "null";
74 }
75 else
76 {
77 // Generate a unique key for the particle
78 std::ostringstream key_stream;
79 key_stream << particle->get_name() << "_" << particle_count[particle->get_name()] + 1;
80 key = key_stream.str();
81 particle_type = particle->get_type();
82 }
83
84 // Insert the particle into the multimap with the generated key
85 particle_container.insert({key, {particle, particle_type}});
86
87 // Increment the count for this particle type
88 particle_count[particle->get_name()]++;
89}
